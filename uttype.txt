private let ndjson: UTType = {
    let type = UTType(
        tag: "ndjson",
        tagClass: .filenameExtension,
        conformingTo: .text
    )

    print("UTType ndjson:", type as Any)
    print("preferred ext:", type?.preferredFilenameExtension as Any)
    print("identifier:", type?.identifier as Any)

    return type ?? .text
}()
